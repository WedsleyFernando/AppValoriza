// Dependencias 
	
 - Ionic
	npm i -g ionic
 - Cordova
	npm i -g cordova  

 - Firebase
	npm install @ionic-native/firebase
	ionic cordova plugin add cordova-plugin-firebase
	
 - RXJS
	npm install rxjs
	
 - Storage 
	npm install --save @ionic/storage
	ionic cordova plugin add cordova-sqlite-storage
	
 - AngularFire2
	npm install angularfire2
